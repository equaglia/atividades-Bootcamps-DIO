fun soma(a:Int, b:Int) = a.plus(b)
fun subtrai(a:Int, b:Int) = a.minus(b)
fun multiplica(a:Int, b:Int) = a.times(b)
fun divide(a:Int, b:Int) = a.div(b)

fun calcula(a:Int, b:Int, operacao:(Int,Int)->Int) = operacao(a,b)
            
fun calculadora(a:Int, b:Int, oper:Int):Int{
    val resultado = when(oper){
        SOMAR -> calcula(a,b,::soma)
        SUBTRAIR -> calcula(a,b,::subtrai)
        MULTIPLICAR -> calcula(a,b,::multiplica)
        DIVIDIR -> calcula(a,b,::divide)
        else -> 0
    }
    println(resultado)
    return resultado
}
